<section class="mx-auto mt-24 max-w-3xl px-4">
  <h1 class="mb-4 text-2xl font-semibold text-slate-800">Carte salvate</h1>

  @if (isLoading) {
  <div class="rounded-md border border-slate-200 bg-slate-50 p-4 text-sm text-slate-600">
    Caricamento carte salvate...
  </div>
  } @else if (methods.length === 0) {
  <div class="rounded-md border border-slate-200 bg-white p-4 text-sm text-slate-600">
    Nessuna carta salvata per questo account.
  </div>
  } @else {
  <div class="flex flex-col gap-3">
    @for (method of methods; track method.id) {
    <mat-card appearance="outlined" class="payment-card">
      <mat-card-content class="flex items-center justify-between gap-3">
        <div>
          <p class="font-medium text-slate-800">{{ method.brand }} •••• {{ method.last4 }}</p>
          <p class="text-sm text-slate-600">Scadenza: {{ method.expMonth }}/{{ method.expYear }}</p>
          @if (method.isDefault) {
          <p class="text-xs font-semibold uppercase tracking-wide text-emerald-700">Predefinita</p>
          }
        </div>
        <div class="flex items-center gap-2">
          <button mat-stroked-button [disabled]="isSaving || method.isDefault" (click)="setDefault(method.id)">
            Imposta default
          </button>
          <button mat-stroked-button color="warn" [disabled]="isSaving" (click)="deleteMethod(method.id)">
            Elimina
          </button>
        </div>
      </mat-card-content>
    </mat-card>
    }
  </div>
  }
</section>
